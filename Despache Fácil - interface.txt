    <script>
        // Tab Management
        function showSection(sectionId) {
            // Update Title
            const titles = {
                'dashboard': 'Visão Geral',
                'cotacao': 'Nova Cotação de Frete',
                'envios': 'Gestão de Envios',
                'financeiro': 'Controlo Financeiro'
            };
            document.getElementById('section-title').innerText = titles[sectionId];

            // Toggle visibility
            const sections = ['dashboard', 'cotacao', 'financeiro'];
            sections.forEach(s => {
                const el = document.getElementById('section-' + s);
                if (el) el.classList.add('hidden');
                
                const link = document.getElementById('link-' + s);
                if (link) {
                    link.classList.remove('nav-active');
                    link.classList.add('text-slate-500');
                }
            });

            document.getElementById('section-' + sectionId).classList.remove('hidden');
            document.getElementById('link-' + sectionId).classList.add('nav-active');
            document.getElementById('link-' + sectionId).classList.remove('text-slate-500');
        }

        // Simulate Quote Results
        function simulateQuote() {
            const results = document.getElementById('quote-results');
            results.classList.remove('hidden');
            results.classList.add('animate-in', 'slide-in-from-top-4');
        }

        // Charts Initialization
        window.onload = function() {
            // Main Line Chart
            const ctxLine = document.getElementById('shipmentChart').getContext('2d');
            new Chart(ctxLine, {
                type: 'line',
                data: {
                    labels: ['Jan', 'Fev', 'Mar', 'Abr', 'Mai', 'Jun'],
                    datasets: [{
                        label: 'Envios Totais',
                        data: [150, 230, 180, 290, 250, 320],
                        borderColor: '#3b82f6',
                        backgroundColor: 'rgba(59, 130, 246, 0.05)',
                        fill: true,
                        tension: 0.4,
                        borderWidth: 4,
                        pointRadius: 0
                    }]
                },
                options: {
                    responsive: true,
                    plugins: { legend: { display: false } },
                    scales: {
                        y: { border: { display: false }, grid: { color: '#f1f5f9' }, ticks: { font: { family: 'Outfit' } } },
                        x: { grid: { display: false }, ticks: { font: { family: 'Outfit' } } }
                    }
                }
            });

            // Carrier Pie Chart
            const ctxPie = document.getElementById('carrierPieChart').getContext('2d');
            new Chart(ctxPie, {
                type: 'doughnut',
                data: {
                    labels: ['Correios', 'Jadlog', 'Total Express'],
